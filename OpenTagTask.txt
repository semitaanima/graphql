query getAllSpeciesFromNewHope {
  allFilms(first: 1) {
  pageInfo{startCursor},
    films{title},
    edges{
      node{
        speciesConnection{
          species{
            name, classification
          },
          totalCount
        }
      }
    }
      }
    }

query getTotalPeopleCountFromNewHope {
  allFilms(first: 1) {
  pageInfo{startCursor},
    films{title},
    edges{
      node{
        characterConnection{
          characters{
            id, name, gender
          },
          totalCount
          }
        }
      }
   } 
 }
